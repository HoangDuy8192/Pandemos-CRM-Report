function fmt(n) { if(n>=1e9)return(n/1e9).toFixed(1)+'B'; if(n>=1e6)return(n/1e6).toFixed(1)+'M'; if(n>=1e3)return(n/1e3).toFixed(0)+'K'; return n; }
function fmtVND(n) { return new Intl.NumberFormat('vi-VN').format(Math.round(n)); }
function pct(a,b) { return b>0?(a/b*100).toFixed(1)+'%':'—'; }

// ── Fetch data ──
async function loadData() {
  try {
    document.getElementById('loadingText').textContent = 'Đang kết nối GitHub...';
    const res = await fetch(DATA_URL + '?t=' + Date.now());
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    D = await res.json();
    document.getElementById('loadingText').textContent = 'Đang render dashboard...';
    renderAll();
    document.getElementById('loadingOverlay').style.display = 'none';
  } catch(e) {
    document.getElementById('loadingText').innerHTML = `
      <span style="color:#e89a90">❌ Không tải được data.json</span><br>
      <span style="font-size:11px;color:var(--text-muted)">Lỗi: ${e.message}</span><br><br>
      <a href="upload.html" style="color:var(--gold);font-size:12px">→ Upload data lần đầu tại đây</a>
    `;
    document.querySelector('.spinner').style.display = 'none';
  }
}

function renderValue() {

  document.getElementById('v-totalRevSub').textContent = 'Lũy kế toàn bộ (VND)';

  const prov = p.topProvinces;

function buildNUTrendChart() {
  const nu = D.nearUpgrade;
  const src = nu[nuActiveView];
  const ds = [];
  if (nuActiveTier==='all'||nuActiveTier==='silver') ds.push({label:'Member→Silver',data:src.silver,borderColor:C.SILVER,backgroundColor:'rgba(158,166,180,.08)',pointRadius:3,tension:.3,fill:false,borderWidth:2});
  if (nuActiveTier==='all'||nuActiveTier==='gold')   ds.push({label:'Silver→Gold',  data:src.gold,  borderColor:C.GOLD,  backgroundColor:'rgba(201,168,76,.08)',  pointRadius:3,tension:.3,fill:false,borderWidth:2});
  if (nuActiveTier==='all'||nuActiveTier==='plat')   ds.push({label:'Gold→Platinum',data:src.plat,  borderColor:C.PLAT,  backgroundColor:'rgba(212,208,200,.08)',  pointRadius:3,tension:.3,fill:false,borderWidth:2});
  if (nuTrendChart) nuTrendChart.destroy();
  nuTrendChart = new Chart(document.getElementById('chartNUTrend'), {
    type:'line', data:{labels:src.labels,datasets:ds},
    options:{ plugins:{legend:{display:true,labels:{color:'#8a8278',boxWidth:12,padding:14}},tooltip:{callbacks:{label:c=>`${c.dataset.label}: ${c.parsed.y} KH`}}}, scales:{y:{grid:{color:'#2e2b26'},ticks:{color:'#5a5650'}},x:{grid:{display:false}}} }
  });
}
